### UserChat

For tournaments, simuls (requires login):

```scala
case class UserChat(
    id: ChatId,
    lines: List[UserLine]
) extends Chat[UserLine]:

  val loginRequired = true

  def forMe(using me: Option[Me], all: AllMessages): UserChat =
    if all.yes || me.exists(_.marks.troll) then this
    else copy(lines = lines.filterNot(_.troll))

  def markDeleted(u: User) = copy(
    lines = lines.map: l =>
      if l.userId.is(u.id) then l.delete else l
  )

  def hasLinesOf(u: User) = lines.exists(_.userId == u.id)

  def add(line: UserLine) = copy(lines = lines :+ line)

  def mapLines(f: UserLine => UserLine) = copy(lines = lines.map(f))

  def filterLines(f: UserLine => Boolean) = copy(lines = lines.filter(f))

  def userIds = lines.view.map(_.userId).toSet

  def truncate(max: Int) = copy(lines = lines.drop((lines.size - max).atLeast(0)))

  def hasRecentLine(u: User): Boolean = 
    lines.reverse.take(12).exists(_.userId == u.id)
```

### MixedChat

For games (allows anonymous players):

```scala
case class MixedChat(
    id: ChatId,
    lines: List[Line]
) extends Chat[Line]:

  val loginRequired = false

  def forMe(using me: Option[Me], all: AllMessages): MixedChat =
    if all.yes || me.exists(_.marks.troll) then this
    else
      copy(lines = lines.filter:
        case l: UserLine => !l.troll
        case _: PlayerLine => true)

  def mapLines(f: Line => Line) = copy(lines = lines.map(f))

  def userIds = lines
    .collect:
      case l: UserLine => l.userId
    .toSet
```

```scala
// Base trait
sealed trait Line:
  def text: String
  def isHuman: Boolean
  def isSystem = !isHuman
  def delete: Line

// User message (logged-in user)
case class UserLine(
    username: UserName,
    title: Option[PlayerTitle],
    patron: Boolean,
    text: String,
    troll: Boolean,
    deleted: Boolean
) extends Line:
  def isHuman = !deleted
  def userId = username.id
  def delete = copy(deleted = true)

// Player message (anonymous in games)
case class PlayerLine(
    color: Color,
    text: String
) extends Line:
  def isHuman = true
  def delete = this  // Can't delete player lines

// System message
case class SystemLine(
    text: String
) extends Line:
  def isHuman = false
  def delete = this
```

## JSON View

From `modules/chat/src/main/ChatJsonView.scala`:

```scala
final class ChatJsonView(
    lightUserApi: lila.user.LightUserApi
)(using Executor):
  
  def apply(
      chat: AnyChat,
      writeable: Boolean = true
  )(using me: Option[Me]): Fu[JsObject] =
    chat match
      case user: UserChat => userChat(user, writeable)
      case mixed: MixedChat => mixedChat(mixed, writeable)
  
  private def lineJson(line: Line): Fu[JsObject] = line match
    case UserLine(username, title, patron, text, troll, deleted) =>
      Json.obj(
        "u" -> username,
        "t" -> text,
        "d" -> deleted.option(true)
      )
    case PlayerLine(color, text) =>
      Json.obj(
        "c" -> color.name,
        "t" -> text
      )
    case SystemLine(text) =>
      Json.obj("t" -> text)
```

## Chat Features by Context

- Presets available
- Can be restricted
- Linked to game result

- User only (login required)
- Public channel
- Moderated
- Tournament announcements

- User only
- Permissions based on study settings
- Can be disabled
- Linked to study positions

### Team Chat
- User only (team members)
- Persistent history
- Team announcements
- Moderated by team leaders

## Performance

1. **Caching** - Recent chats cached in memory
2. **Truncation** - Limit lines per chat
3. **Lazy loading** - Load on demand
4. **Batch updates** - Group WebSocket sends
5. **Indexed queries** - Fast chat lookup
