The **study** module enables collaborative chess analysis through studies - collections of annotated game variations, opening preparations, and shared chess content.

## Core Types

### Study

From `modules/study/src/main/Study.scala`:

```scala
case class Study(
    @Key("_id") id: StudyId,
    name: StudyName,
    members: StudyMembers,
    position: Position.Ref,              // Current chapter and path
    ownerId: UserId,
    visibility: Visibility,
    settings: Settings,
    from: Study.From,                    // Source (game, scratch, etc.)
    likes: Study.Likes,
    description: Option[String] = None,
    topics: Option[StudyTopics] = None,
    flair: Option[Flair] = None,
    createdAt: Instant,
    updatedAt: Instant
) extends hub.Study:

  val slug = scalalib.StringOps.slug(name.value)

  def owner = members.get(ownerId)

  def isOwner[U: UserIdOf](u: U) = ownerId.is(u)

  def isMember[U: UserIdOf](u: U) = members.contains(u.id)

  def canChat(id: UserId) = 
    Settings.UserSelection.allows(settings.chat, this, id.some)

  def canContribute[U: UserIdOf](u: U): Boolean =
    isOwner(u) || members.get(u.id).exists(_.canContribute) || u.is(UserId.lichess)

  def canView(id: Option[UserId]) = !isPrivate || id.exists(members.contains)

  def isCurrent(c: Chapter.Like) = c.id == position.chapterId

  def withChapter(c: Chapter.Like): Study = 
    if isCurrent(c) then this else rewindTo(c.id)

  def rewindTo(chapterId: StudyChapterId): Study =
    copy(position = Position.Ref(chapterId = chapterId, path = UciPath.root))

  def isPublic = visibility == Visibility.public
  def isUnlisted = visibility == Visibility.unlisted
  def isPrivate = visibility == Visibility.`private`

  def isNew = (nowSeconds - createdAt.toSeconds) < 4

  def isOld = (nowSeconds - updatedAt.toSeconds) > 20 * 60

  def isRelay = from match
    case _: From.Relay => true
    case _ => false

  def notable = likes.value > 10 || {
    likes.value > 5 && createdAt.isBefore(nowInstant.minusDays(7))
  }

  def cloneFor(user: User): Study =
    val owner = StudyMember(id = user.id, role = StudyMember.Role.Write)
    copy(
      id = Study.makeId,
      members = StudyMembers(Map(user.id -> owner)),
      ownerId = owner.id,
      visibility = Visibility.`private`,
      from = Study.From.Study(id),
      likes = Likes(1),
      createdAt = nowInstant,
      updatedAt = nowInstant
    )
```

## Study Members

```scala
case class StudyMembers(members: Map[UserId, StudyMember]):
  def get(userId: UserId): Option[StudyMember] = members.get(userId)
  
  def contains(userId: UserId): Boolean = members.contains(userId)
  
  def contributorIds: Set[UserId] = 
    members.collect:
      case (id, m) if m.canContribute => id
    .toSet

case class StudyMember(
    id: UserId,
    role: StudyMember.Role,
    addedAt: Instant = nowInstant
):
  def canContribute = role >= StudyMember.Role.Write

object StudyMember:
  enum Role:
    case Read
    case Write
    
  given Ordering[Role] = Ordering.by(_.ordinal)
```

## Study Settings

```scala
// From modules/study/src/main/Settings.scala
case class Settings(
    computer: Settings.UserSelection,
    explorer: Settings.UserSelection,
    cloneable: Settings.UserSelection,
    shareable: Settings.UserSelection,
    chat: Settings.UserSelection,
    sticky: Boolean,
    description: Boolean
)

object Settings:
  enum UserSelection:
    case Everyone
    case Members
    case Contributors  
    case Nobody
    
  object UserSelection:
    def allows(
        selection: UserSelection, 
        study: Study, 
        userId: Option[UserId]
    ): Boolean = selection match
      case Everyone => true
      case Members => userId.exists(study.isMember)
      case Contributors => userId.exists(study.canContribute)
      case Nobody => false
```

## Real-time Collaboration

Studies support real-time collaborative editing:

1. **WebSocket updates** - Live position sync
2. **Member presence** - See who's viewing
3. **Concurrent editing** - Multiple contributors
4. **Comment threads** - Discussion on positions
5. **Shape drawing** - Visual annotations
